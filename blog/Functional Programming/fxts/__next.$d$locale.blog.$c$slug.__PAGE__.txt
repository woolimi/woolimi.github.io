```js
const getFlatUser = pipeLazy(

  setAddress,

  isOwner,

  setPersonalInfo,
  setAddress,
  setHouseInfo,
  setOutbuildingSize,

const flatUser = getFlatUser();
```

```js
const EMAIL_REGEX =
  /^(([^<>()[\]\\.,;:\s@"]+(\.[^<>()[\]\\.,;:\s@"]+)*)|(".+"))@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\])|(([a-zA-Z\-0-9]+\.)+[a-zA-Z]{2,}))$/;
const EmailAPI = {
  async check(email) {
    const isValid = !!email.match(EMAIL_REGEX);
    await delay(500);
    return isValid ? { email, format_valid: true } : { email, format_valid: false };
  },
};

const emails = [
  'Tee@gmail.com',
  'Tom@gmail.com',
  'Tommy@gmail.com',
  'Mas@gmail.com',
  'this.is.not.valid.email.com', // Invalid email
  'Moz@gmail.com',
  'Thom@gmail.com',
  'Tomzy@gmail.com',
  'Tizzy@gmail.com',
  'T-mas@gmail.com',
  'Mars@gmail.com',
];

const checkEmail = pipeLazy(
  toAsync,
  map(EmailAPI.check),
  concurrent(3),
  peek(console.log),
  toArray,
);
await checkEmail(emails);
```
