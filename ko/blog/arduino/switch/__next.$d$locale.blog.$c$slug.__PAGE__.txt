아두이노를 사용하면서 스위치(버튼)을 제어할때 자주 사용하는 코드를 정리.

## 버튼을 1번만 인식하기

버튼은 눌린 동안 계속 HIGH 신호를 보내기 때문에, 한 번 누른 것을 여러 번 누른 것으로 인식하는 문제가 발생한다. 따라서 이전 상태를 저장하고 해당 상태가 LOW -> HIGH 로 바뀔때만 true 를 리턴하는 함수를 만들었다.

```c++
bool isButtonPressed(int pin) {
  static int prevState = LOW;
  int currentState = digitalRead(pin);

	if (currentState == HIGH && prevState == LOW) {
		prevState = HIGH;
		return true;
	}

	prevState = currentState;
  return false;
}
```

## 버튼 1번 + debounce

버튼을 사용하다보면 기계적 접점의 바운싱(떨림) 현상으로 인해 노이즈가 발생할 수 있다. 이를 해결하기 위해 debounce (50ms) 를 이용해 신호변화가 안정화될때까지 delay 시킨다.

```c++
bool isButtonPressed(int pin) {
  static int prevState = LOW;
  static unsigned long lastDebounceTime = 0;
  const unsigned long DEBOUNCE_DELAY = 50;  // 50ms

  int currentState = digitalRead(pin);
  unsigned long currentTime = millis();

  // 상태가 변했으면 타이머 리셋
  if (currentState != prevState) {
    lastDebounceTime = currentTime;
  }

  // debounce 시간이 지났고, rising edge인 경우
  if ((currentTime - lastDebounceTime) > DEBOUNCE_DELAY) {
    static int stableState = LOW;

    if (currentState != stableState) {
      stableState = currentState;

      if (stableState == HIGH) {
        prevState = currentState;
        return true;  // 버튼 눌림 감지!
      }
    }
  }

  prevState = currentState;
  return false;
}
```

## 버튼이 여러개인경우

```c++
class Button {
public:
  Button(int pin) : pin(pin), prevState(LOW) {
    pinMode(pin, INPUT_PULLUP);
  }

  bool isPressed() {
    int currentState = digitalRead(pin);

    // 눌림(HIGH로 변했을 때) 감지
    if (currentState == HIGH && prevState == LOW) {
      prevState = currentState;
      return true;
    }

    prevState = currentState;
    return false;
  }

private:
  int pin;
  int prevState;
};
```
